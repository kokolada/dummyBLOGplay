create database CDOM  // collation latin_general_ci

CREATE TABLE korisnik(
    id INT AUTO_INCREMENT PRIMARY KEY,
    korisnickoime VARCHAR(30),
    lozinka VARCHAR(60),
    remember_token VARCHAR(100),
    created_at timestamp,
    updated_at timestamp
    )

CREATE TABLE Vijesti(
	id INT AUTO_INCREMENT PRIMARY KEY,
	naslov VARCHAR(250),
    slika VARCHAR(100),
	sadrzaj MEDIUMTEXT,
    created_at timestamp,
    updated_at timestamp,
    korisnik_id int,
    foreign key(korisnik_id) references korisnik(id)
)

CREATE TABLE baneri(
	id INT AUTO_INCREMENT PRIMARY KEY,
	naziv VARCHAR(20),
	slika VARCHAR(100),
	tip tinyint
)

CREATE TABLE REKLAME(
	id INT AUTO_INCREMENT PRIMARY KEY,
	broj int,
	slika VARCHAR(100)
)

CREATE TABLE PARTNERI(
	id INT AUTO_INCREMENT PRIMARY KEY,
	naziv VARCHAR(40),
	sajt VARCHAR(100),
	slika VARCHAR(100),
	opartneru text,
	pogodnosti VARCHAR(250),
	grupa int,
	created_at timestamp,
	updated_at timestamp    
)

create table logo(
    id int auto_increment primary key,
    link varchar(100)
)

create table ocdomcard(
 	id int auto_increment primary key,
    slika varchar(100),
    opis text
)
